#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
void sort(int arr[], int n) {
    int i, j, temp;
    for(i=0; i<n-1; i++) {
        for(j=i+1; j<n; j++) {
            if(arr[i] > arr[j]) 
			{
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}

int main() {
    int arr[5], i;
    printf("Enter 5 numbers: ");
    for(i=0; i<5; i++) {
        scanf("%d", &arr[i]);
    }

    int pid = fork();

    if(pid == 0) { 
        printf("\nChild Process\n");
        sort(arr, 5);
        printf("Child sorted array: ");
        for(i=0; i<5; i++) {
            printf("%d ", arr[i]);
        }
        printf("\n");
    }
    else { 
        wait(NULL); 
        printf("\nParent Process\n");
        sort(arr, 5);
        printf("Parent sorted array: ");
        for(i=0; i<5; i++) {
            printf("%d ", arr[i]);
        }
        printf("\n");
    }
    return 0;
}
